# 피보나치 수열

```swift
func pibonachi(_ num: Int) -> Int {
    if num == 1 {
        return 1
    } else if num == 2 {
        return 1
    }

    return pibonachi(num - 1) + pibonachi(num - 2)
}
```
- 이렇게 풀면 재귀함수 엄청 오래 걸린다


```swift

import Foundation

func pibonachi(_ num: Int) -> Int {
    var numberArray = Array(repeating: 0, count: 100)
    if num == 1 {
        return 1
    }
    if num == 2 {
        return 1
    }

    if numberArray[num] != 0 {
        return numberArray[num]
    }

    numberArray[num] = pibonachi(num - 1) + pibonachi(num - 2)

    return numberArray[num]
}


print(pibonachi(30))
 
```

### 다음에 올 숫자

```swift
func solution(_ common:[Int]) -> Int {
    guard common.min()! > -1000,
          common.max()! < 2000 else { return 0 }
    
    guard common.count > 2,
          common.count < 1000 else { return 0 }
    
    if common[2] - common[1] == common[1] - common[0] {
        let d = common[2] - common[1]
        return common.last! + d
    }
    
    let r = common[1] / common[0]
    return common.last! * r
}
```
